% Formalización en Paraguay

\section{¿Por qué formalizarse?}

\begin{itemize}
\item \textbf{Facturación legal}: Podés vender a empresas, gobiernos, ferias formales, etc.
\item \textbf{Acceso a créditos y subsidios}: CAH, Crédito Ñepyrura, FOMIPYMES
\item \textbf{Participar en ferias oficiales}: Del SNPP, MIC, municipios o embajadas
\item \textbf{Protecciones legales}: Sos reconocido como proveedor formal
\item \textbf{Posibilidad de emitir factura} o comprobante simple
\end{itemize}

\begin{calloutnote}
\textbf{Ejemplo cercano}: Una artesana formalizada fue seleccionada para una feria de la SENATUR en el exterior, porque tenía RUC y estaba inscripta en el Registro de MIPYMES.
\end{calloutnote}

\section{Pasos básicos para formalizarse}

\subsection{Obtener RUC simplificado}

\begin{itemize}
\item Se hace en la \textbf{DNIT} (Dirección Nacional de Ingresos Tributarios) o vía web.
\item Solo necesitás:
  \begin{itemize}
  \item Cédula
  \item Comprobante de dirección (puede ser declaración jurada)
  \item Declarar actividad económica
  \end{itemize}
\end{itemize}

\textbf{Régime Resimple:} Solo una vez al año, no necesitas contador.

\subsection{Inscripción como MIPYME (MIC)}

Se hace en el Ministerio de Industria y Comercio (MIC)

\subsubsection{Ventajas:}

\begin{itemize}
\item Prioridad en programas de apoyo
\item Certificación como proveedor MIPYME
\item Acceso a capacitaciones y redes
\end{itemize}

\begin{callouttip}
Se puede hacer online: \url{https://www.mic.gov.py/}
\end{callouttip}
