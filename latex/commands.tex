% Custom callout environments

% TIP callout (green border)
\newenvironment{callouttip}[1][Consejo]{%
  \begin{mdframed}[
    linecolor=green!60!black,
    linewidth=2pt,
    leftmargin=0pt,
    rightmargin=0pt,
    innerleftmargin=10pt,
    innerrightmargin=10pt,
    innertopmargin=8pt,
    innerbottommargin=8pt,
    skipabove=10pt,
    skipbelow=10pt,
    frametitle={\textbf{#1}},
    frametitlefont=\bfseries\sffamily
  ]
}{%
  \end{mdframed}
}
